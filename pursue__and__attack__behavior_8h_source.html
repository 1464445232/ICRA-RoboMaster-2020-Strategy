49  double my_robot_x = 0, my_robot_y = 0, enemy_robot_x = 0, enemy_robot_y = 0;
50 
51  my_robot_x = this->p_my_robot_->GetChassisMapPose().pose.position.x;

56  enemy_robot_y = this->p_blackboard_->GetEnemyRobot1().GetPose().pose.position.y;

63

68  void getGoalPose(RobotId enemy_id, double aim_distance, geometry_msgs::PoseStamped &pose_stamped) {
69  double my_robot_x = 0, my_robot_y = 0, enemy_robot_x = 0, enemy_robot_y = 0;

71  my_robot_x = this->p_my_robot_->GetChassisMapPose().pose.position.x;
72  my_robot_y = this->p_my_robot_->GetChassisMapPose().pose.position.y;
73 
74  if (enemy_id == ENEMY_ROBOT_1) {
75  enemy_robot_x = this->p_blackboard_->GetEnemyRobot1().GetPose().pose.position.x;
76  enemy_robot_y = this->p_blackboard_->GetEnemyRobot1().GetPose().pose.position.y;
77  } else if (enemy_id == ENEMY_ROBOT_2) {
78  enemy_robot_x = this->p_blackboard_->GetEnemyRobot2().GetPose().pose.position.x;
79  enemy_robot_y = this->p_blackboard_->GetEnemyRobot2().GetPose().pose.position.y;
80  } else {
81  ROS_ERROR("PursueAndAttack get the error enemy_robot_id !");
82  }
83 
84  double distance = 0.0;
85  distance = getDistance(enemy_id);
86 
87  pose_stamped.pose.position.x = my_robot_x + (distance - aim_distance) / distance * (enemy_robot_x - my_robot_x);
88  pose_stamped.pose.position.y = my_robot_y + (distance - aim_distance) / distance * (enemy_robot_y - my_robot_y);
89 
90  if (enemy_robot_x - my_robot_x < 0) {
91  pose_stamped.pose.orientation.w = tf::createQuaternionFromYaw(M_PI - (enemy_robot_y - my_robot_y) / distance).w();
92  pose_stamped.pose.orientation.x = tf::createQuaternionFromYaw(M_PI - (enemy_robot_y - my_robot_y) / distance).x();
93  pose_stamped.pose.orientation.y = tf::createQuaternionFromYaw(M_PI - (enemy_robot_y - my_robot_y) / distance).y();
94  pose_stamped.pose.orientation.z = tf::createQuaternionFromYaw(M_PI - (enemy_robot_y - my_robot_y) / distance).z();
95  } else {
96  pose_stamped.pose.orientation.w = tf::createQuaternionFromYaw((enemy_robot_y - my_robot_y) / distance).w();
97  pose_stamped.pose.orientation.x = tf::createQuaternionFromYaw((enemy_robot_y - my_robot_y) / distance).x();
98  pose_stamped.pose.orientation.y = tf::createQuaternionFromYaw((enemy_robot_y - my_robot_y) / distance).y();
99  pose_stamped.pose.orientation.z = tf::createQuaternionFromYaw((enemy_robot_y - my_robot_y) / distance).z();
100  }
